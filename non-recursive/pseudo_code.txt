Algorithm count_distinct(array, n):
    for i ← 0 to n-1 do
        for j ← i+1 to n-1 do
            if array[j] < array[i] then
                SWAP(array[i], array[j])

    count ← 0

    for i ← 0 to n-1 do
        if i == 0 OR array[i] ≠ array[i-1] then
            count ← count + 1

    return count
